!
! cisco autoinstall with dhcp option 67
!
! Date: 2022-11-03
!
! SERVERIP = 192.168.10.15
! PROTOCOLL = HTTP
! IMAGENAME = c1100-universalk9.17.06.03a.SPA.bin
! DSTINATIONFLASH = flash
! USERNAME = update
! PASSWORD = update
!
loggin console emergencies
!loggin console debug
!
clock timezone CET 1 0
clock summer-time CEST recurring last Sun Mar 2:00 last Sun Oct 3:00
!
line con 0
  logging synchronous
  width 150
exit
!
do send log 0 STARTING AUTOINSTALL
!
do send log 0 disable file prompts
file prompt quiet
!
do send log 0 copy image <IMAGENAME> from <PROTOCOL>  to flash
do copy <PROTOCOL>://<USERNAME>:<PASSWORD>@<SERVERIP>/<IMAGENAME> <DSTINATIONFLASH>:/<IMAGENAME>
!
do send log 0 clear boot variable
no boot system
!
do send log 0 configure new boot variable
boot system flash <IMAGENAME>
!
do send log 0 creating eem applet reload
event manager applet reload
 description reload system
 event none maxrun 60
 action 10 cli command "enable"
 action 20 cli command "send log 0 EEM STARTING RELOAD"
 action 30 cli command "reload" pattern "\[confirm\]"
 action 30.1 cli command "y"
 action 50 cli command "send log 0 EEM FINISH RELOAD"
 exit
!
do send log 0 creating eem applet to cleanup on next reload
event manager applet cleanup
 description clear all config after reload
 event timer cron cron-entry "@reboot" maxrun 600 name Cleanup
 action 1 cli command "enable"
 action 2 cli command "send log 0 EEM STARTING CLEANUP"
 action 3 wait 120
 action 40.1 cli command "send log 0 CLEAR CONFIG"
 action 40.2 cli command "write erase" pattern "\[confirm\]"
 action 40.3 cli command "y"
 action 50.1 cli command "send log 0 REMOVE UNUSED IMAGES"
 action 50.2 cli command "install remove inactive" pattern ".* \[y/n\]"
 action 50.3 cli command "y"
 action 999 cli command "send log 0 EEM FINISH CLEANUP"
 exit
!
do send log 0 save config
do wr mem
!
do send log 0 reload device
do event manager run reload
!
do send log 0 FINISH autoinstall going to reboot
end
